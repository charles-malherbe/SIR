#include <iostream>

using namespace std;

int main()
{
    double H = 500000;
    double Z = 1;
    double M = 0;
    double I = 0;
    
    double H1, Z1, M1, I1;
    
    double this->alpha = 0.0095; // coeff Humain -> Infecté
    double this->beta = 0.005; // coeff Infecté -> Zombie
    double this->epsilon = 0.0001; // coeff Humain                                                 -> Mort
    double this->gamma = 0.0001; // coeff Infecté -> Mort
    double this->delta = 0.0001; // coeff Mort -> Zombie
    double this->zeta = 0.005; // coeff Zombie -> Mort
    double this->eta = 0.0045; // coeff Zombie -> Humain
    
    int compte = 0;
    
    while (int(H) > 0) {
        
        //Pour le calcul j'ai considéré que le taux de naissance était de 0
        
        H1 = (0 - H*Z*this->alpha - H*this->epsilon + Z*this->eta)*0.0001;  //calcul de la différence de population d'Humains sur un très court intervalle de temps
        I1 = (H*Z*this->alpha - this->beta*I- this->gamma*I)*0.0001; //calcul de la différence de population d'Infectés sur un très court intervalle de temps
        Z1 = (this->beta*I + M*this->delta - Z*H*this->zeta - Z*this->eta)*0.0001; //calcul de la différence de population de Zombies sur un très court intervalle de temps
        M1 = (H*this->epsilon + this->gamma*I + H*Z*this->zeta - M*this->delta)*0.0001; //calcul de la différence de population de Morts sur un très court intervalle de temps
        
        //On met à jour le nombre d'Humains / Zombies / Infectés / Morts
        
        H = H+H1;
        Z = Z+Z1;
        M = M+M1;
        I = I+I1;
        
        compte++;
    }
    cout<<" Humains : "<<int(H)<<"\n Infectés : "<<int(I)<<"\n Zombies : "<<int(Z)<<"\n Morts : "<<int(M);
    cout<<"\n Il n'y a plus aucun Humain sain après la "<<compte<<"eme iterations";

    return 0;
}
